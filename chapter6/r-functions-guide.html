Beschrijvende Statistieken in R: Complete Functiegids

Overzicht van de belangrijkste R functies en packages voor beschrijvende statistieken, met praktische voorbeelden en vergelijkingen.

Basis R Functies

Centraliteitsmaten

# Gemiddelde
mean(data$variabele, na.rm = TRUE)

# Mediaan
median(data$variabele, na.rm = TRUE)

# Modus (handmatige functie)
calculate_mode <- function(x) {
  unique_vals <- unique(x[!is.na(x)])
  tabulated <- tabulate(match(x, unique_vals))
  unique_vals[which.max(tabulated)]
}
calculate_mode(data$variabele)

Spreidingsmaten

# Standaarddeviatie
sd(data$variabele, na.rm = TRUE)

# Variantie
var(data$variabele, na.rm = TRUE)

# Bereik (range)
range(data$variabele, na.rm = TRUE)
diff(range(data$variabele, na.rm = TRUE))  # Bereik als enkel getal

# Interkwartielafstand
IQR(data$variabele, na.rm = TRUE)

# Kwartielen
quantile(data$variabele, probs = c(0.25, 0.50, 0.75), na.rm = TRUE)

Samenvatting

# Basis samenvatting
summary(data$variabele)
# Output: Min, Q1, Median, Mean, Q3, Max, NA's

# Voor meerdere variabelen
summary(data[, c("var1", "var2", "var3")])

psych::describe() - Meest Compleet

De psych package biedt de meest uitgebreide functie voor beschrijvende statistieken.

Installatie en Laden

# Installeren (eenmalig)
install.packages("psych")

# Laden (elke sessie)
library(psych)

Basis Gebruik

# Eén variabele
describe(data_wave1$W1_Relstress1_Gem)

# Meerdere variabelen
describe(data_wave1[, c("W1_Relstress1_Gem", 
                         "W1_Werkstress_Gem",
                         "W1_Financstress_Gem")])

# Hele dataset
describe(data_wave1)

Nuttige Opties

# Fast = TRUE voor snellere berekening (geen skew/kurtosis)
describe(data, fast = TRUE)

# Omit = TRUE om missing values te verwijderen
describe(data, omit = TRUE)

# Quant = c(.25, .75) voor specifieke kwartielen
describe(data, quant = c(.25, .75))

# IQR = TRUE voor interkwartielafstand
describe(data, IQR = TRUE)

skimr::skim() - Visueel en Uitgebreid

De skimr package geeft een uitgebreid overzicht met mini-histogrammen.

Installatie en Gebruik

# Installeren
install.packages("skimr")

# Laden
library(skimr)

# Hele dataset bekijken
skim(data_wave1)

# Specifieke variabelen
data_wave1 %>%
  select(W1_Relstress1_Gem, W1_Werkstress_Gem, W1_Financstress_Gem) %>%
  skim()

Voordelen van skimr

  • ✅ Visuele histogrammen: Zie verdeling direct in console
  • ✅ Automatische type detectie: Verschillende statistieken per datatype
  • ✅ Missing value overzicht: Aantal en percentage NA's per variabele
  • ✅ Compacte output: Alles op één scherm

dplyr - Flexibel en Krachtig

De dplyr package is ideaal voor groepsgewijze statistieken.

Basis Functies

library(dplyr)

# Eén variabele, meerdere statistieken
data_wave1 %>%
  summarise(
    n = n(),
    gemiddelde = mean(W1_Relstress1_Gem, na.rm = TRUE),
    mediaan = median(W1_Relstress1_Gem, na.rm = TRUE),
    sd = sd(W1_Relstress1_Gem, na.rm = TRUE),
    min = min(W1_Relstress1_Gem, na.rm = TRUE),
    max = max(W1_Relstress1_Gem, na.rm = TRUE)
  )

Groepsgewijze Statistieken

# Per geslacht
data_wave1 %>%
  group_by(geslacht) %>%
  summarise(
    n = n(),
    M = mean(W1_Relstress1_Gem, na.rm = TRUE),
    SD = sd(W1_Relstress1_Gem, na.rm = TRUE)
  )

# Output:
# geslacht     n     M    SD
# Man         75  3.12  0.89
# Vrouw       75  3.67  0.74

Across() voor Meerdere Variabelen

# Zelfde statistieken voor meerdere variabelen
data_wave1 %>%
  summarise(across(
    c(W1_Relstress1_Gem, W1_Werkstress_Gem, W1_Financstress_Gem),
    list(
      gemiddelde = ~mean(., na.rm = TRUE),
      sd = ~sd(., na.rm = TRUE),
      mediaan = ~median(., na.rm = TRUE)
    )
  ))

# Per groep
data_wave1 %>%
  group_by(geslacht) %>%
  summarise(across(
    c(W1_Relstress1_Gem, W1_Werkstress_Gem),
    list(M = ~mean(., na.rm = TRUE), SD = ~sd(., na.rm = TRUE))
  ))

Custom Functie: Alles-in-Eén

complete_descriptives <- function(data, var_name) {
  library(psych)
  
  # Basis statistieken
  desc <- describe(data[[var_name]])
  
  # 95% BI
  ci <- t.test(data[[var_name]])$conf.int
  
  # Shapiro-Wilk (als n <= 5000)
  shapiro_p <- ifelse(desc$n <= 5000,
                      shapiro.test(data[[var_name]])$p.value,
                      NA)
  
  # Output maken
  cat("\n=== Beschrijvende Statistieken:", var_name, "===\n\n")
  cat("N:", desc$n, "\n")
  cat("M:", round(desc$mean, 2), "\n")
  cat("SD:", round(desc$sd, 2), "\n")
  cat("Mdn:", round(desc$median, 2), "\n")
  cat("IQR:", round(IQR(data[[var_name]], na.rm = TRUE), 2), "\n")
  cat("Min:", round(desc$min, 2), "\n")
  cat("Max:", round(desc$max, 2), "\n")
  cat("Scheefheid:", round(desc$skew, 2), "\n")
  cat("Kurtosis:", round(desc$kurtosis, 2), "\n")
  cat("95% BI: [", round(ci[1], 2), ",", round(ci[2], 2), "]\n")
  
  if (!is.na(shapiro_p)) {
    cat("Shapiro-Wilk p:", round(shapiro_p, 3), "\n")
  }
  
  # Histogram
  hist(data[[var_name]],
       main = paste("Histogram:", var_name),
       xlab = var_name,
       col = "lightblue",
       breaks = 20)
}

# Gebruik
complete_descriptives(data_wave1, "W1_Relstress1_Gem")
